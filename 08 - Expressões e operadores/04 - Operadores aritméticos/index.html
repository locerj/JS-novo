Operadores aritméticos

Multiplicação(*)

console.log(4 * 4)

Divisão(/)

console.log(50 / 4)

Soma(+)

console.log(25 + 32)

Subtração(-)

console.log(7 - 2)

Resto da divisão(%)

console.log(5 % 2)

Incremento(++)

let number = 10

console.log(number++)// incrementar após

console.log(number)

console.log(++number)// incrementar antes

Decremento(--)

console.log(number--)// decrementar após

console.log(number)

console.log(--number)// decrementar antes

Exponencial(**)

console.log(5 ** 2)
