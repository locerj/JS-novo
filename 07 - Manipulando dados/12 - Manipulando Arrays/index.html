Manipulando Arrays

Manipulação de Arrays

let techs = ["html", "css", "js"]

adicionar um itém no fim:

console.log(techs.push('node.js'))

console.log(techs)

adicionar um itém no começo:

console.log(techs.unshift('sql'))

console.log(techs)

remover um itém do fim:

console.log(techs.pop())

console.log(techs)

remover um itém do começo:

console.log(techs.shift())

console.log(techs)

pegar somente alguns elementos do array:

console.log(techs.slice(1, 3))

remover 1 ou mais iténs em qualquer posição do array:

techs.splice(1, 1)

console.log(techs)

techs.splice(1, 2)

console.log(techs)

encontrar a posiçao de um elemento no array:

let index = techs.indexOf("html")

console.log(index)
